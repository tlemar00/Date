012        /*
013        ** Constructor
014        */
015        public Date(int day, int month, int year) throws DateException{
016                
017                /*
018                ** Comprobamos que el año sea correcto
019                */
020                if(year < 1){
021                        throw new DateException("Año " + month + " no valido. El año no puede ser negativo.");
022                }else{
023                        this.year = year;
024                }
025                /*
026                ** Comprobamos que el mes sea correcto
027                */
028                if (month < 1 || month > 12) {
029                        throw new DateException("Mes " + month + " no valido. Valores posibles entre 1 y 12.");
030                } else {
031                        this.month = month;
032                }
033                /*
034                ** Comprobamos que el dia sea correcto
035                */
036                if (day < 1 || day > 31){
037                        throw new DateException("Día " + day + " no válido. Valores posibles entre 1 y 31.");
038                }else{
039                        this.day = day;
040                }
041        }
