001package es.unileon.prg.practicaDate;
002
003public class MainDate {
004
005        public static void main(String[] args) {
006
007                Date today;
008                Date tomorrow;
009
010                try{
011                        today = new Date(18, 03, 2019);
012                        System.out.println("Primera fecha: " + today.toString());
013                        tomorrow = new Date(11, 10, 2019);
014                        System.out.println("Segunda fecha: " + tomorrow.toString());
015                        System.out.println(today.isSameYear(tomorrow));
016                        System.out.println(today.isSameMonth(tomorrow));
017                        System.out.println(today.isSameDay(tomorrow));
018                        System.out.println(today.isSame(tomorrow));
019                        System.out.println("El nombre del mes de " + today + " es: " + today.getMonthName());
020                        System.out.println("El nombre del mes de " + tomorrow + " es: " + tomorrow.getMonthName());
021                        System.out.println("El mes de " + today.getMonthName() + " tiene " + today.getDayOfMonth() + " días.");
022                        System.out.println("El mes de " + tomorrow.getMonthName() + " tiene " + tomorrow.getDayOfMonth() + " días.");
023                        System.out.println("La estación de la primera fecha es: " + today.seasonName());
024                        System.out.println("La estación de la segunda fecha es: " + tomorrow.seasonName());
025                        System.out.println("Desde la primera fecha faltan " + today.getMonthsLeft() + " meses para acabar el año.");
026                        System.out.println("Desde la segunda fecha faltan " + tomorrow.getMonthsLeft() + " meses para acabar el año.");
027                        System.out.println("Desde la primera fecha faltan " + today.daysEnd() + " días para acabar el mes.");
028                        System.out.println("Desde la segunda fecha faltan " + tomorrow.daysEnd() + " días para acabar el mes.");
029                        System.out.println("Los meses con los mismos días que la primera fecha son: " + today.monthsSameDays());
030                        System.out.println("Los meses con los mismos días que la segunda fecha son: " + tomorrow.monthsSameDays());
031                        System.out.println("Días desde principio de año hasta " + today + ": " + today.daysSinceFirst());
032                        System.out.println("Días desde principio de año hasta " + tomorrow + ": " + tomorrow.daysSinceFirst());
033                        System.out.println("Intentos de generar la misma fecha 1 con bucle do-while: " + today.numberAttempsDoWhile());
034                        System.out.println("Intentos de generar la misma fecha 2 con bucle do-while: " + tomorrow.numberAttempsDoWhile());
035                        System.out.println("Intentos de generar la misma fecha 1 con bucle while: " + today.numberAttempsWhile());
036                        System.out.println("Intentos de generar la misma fecha 2 con bucle while: " + tomorrow.numberAttempsWhile());
037                }catch(DateException e){
038                        System.out.println(e.getMessage());
039                }
040                
041        }
042}
